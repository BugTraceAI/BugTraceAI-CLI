```python
class PayloadFeed(Static):
    """Truncate long payloads intelligently."""

    def _truncate_payload(self, payload: str, max_width: int) -> str:
        if len(payload) <= max_width:
            return payload

        # Try to truncate at word boundary
        truncated = payload[:max_width - 3]
        last_space = truncated.rfind(" ")

        if last_space > max_width // 2:
            return truncated[:last_space] + "..."
        else:
            return truncated + "..."

    def render(self) -> Panel:
        width = self.size.width - 4  # Account for padding
        lines = []
        for payload in self.recent_payloads:
            truncated = self._truncate_payload(payload, width)
            lines.append(truncated)
        return Panel("\n".join(lines))
```

### 4. Lazy Loading

```python
class LogInspector(Static):
    """Only render visible logs."""

    MAX_VISIBLE_LINES = 100

    def __init__(self):
        super().__init__()
        self._all_logs = []
        self._visible_start = 0

    def log(self, message: str, level: str) -> None:
        self._all_logs.append((message, level))

        # Auto-scroll if at bottom
        if self._visible_start == len(self._all_logs) - self.MAX_VISIBLE_LINES:
            self._visible_start += 1

    def render(self) -> RenderableType:
        # Only render visible window
        visible_logs = self._all_logs[
            self._visible_start:self._visible_start + self.MAX_VISIBLE_LINES
        ]

        lines = []
        for message, level in visible_logs:
            color = self._get_level_color(level)
            lines.append(f"[{color}]{message}[/{color}]")

        return Panel("\n".join(lines))
```

```python
class ActivityGraph(Static):
    """Debounce high-frequency metrics."""

    def __init__(self):
        super().__init__()
        self._pending_update = False
        self._last_req_rate = 0

    def update_rate(self, rate: float) -> None:
        """Queue update instead of immediate render."""
        self._last_req_rate = rate

        if not self._pending_update:
            self._pending_update = True
            self.set_timer(0.5, self._apply_update)

    def _apply_update(self) -> None:
        """Apply batched update."""
        self.req_rate = self._last_req_rate
        self._pending_update = False
```
